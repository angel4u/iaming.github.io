== 使用systemd优雅启停应用

. 添加依赖 + .pom.xm的build.plugins节点

[source,xml]

org.springframework.boot spring-boot-maven-plugin true

. 添加服务 + ./etc/systemd/system/myapp.service

[source,bash]

[Unit] Description=myapp After=syslog.target

[Service] User=root ExecStart=/var/myapp/myapp.jar SuccessExitStatus=143

[Install]

WantedBy=multi-user.target

. 配置 在jar包同目录下增加同名配置文件 + .myapp.conf

[source,conf]

JAVA_OPTS=“-server -Xms100m -Xmx300m -Xmn128m -Xss256k”

. 启停 +

[source,bash]

systemctl enable myapp # 开机自启动 systemctl status|start|stop|restart myapp

journalctl -u myapp # 看日志，shift+g跳转到最后一行

== 参考 . https://docs.spring.io/spring-boot/docs/current/reference/html/deployment.html
